function speak(text) {

  const speedRate = speed === 0.5 ? 0.24 : speed === 1.2 ? 0.6 : 0.4;

  const url = `https://translate.google.com/translate_tts?ie=UTF-8&q=${encodeURIComponent(text)}&tl=fr&client=tw-ob&ttsspeed=${speedRate}`;

  new Audio(url).play();

}


function playCurrent(){ speak(data[category][index].fr); }
